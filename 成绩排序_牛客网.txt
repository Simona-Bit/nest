        String tempS = names[j];
        int tempG = grades[j];
        names[j] = names[i];
        grades[j] = grades[i];
        names[i] = tempS; 
        grades[i] = tempG;
    }
    public static void sort(int flag){
        for(int i = 1; i < names.length; i++)
            for(int j = i; j > 0 && compare(j,j - 1,flag); j--){

            }
    }
